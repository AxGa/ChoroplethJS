1 ChoroplethJS.GmapOverlay = function() {};
  2 	
  3 ChoroplethJS.GmapOverlay.prototype = new google.maps.OverlayView();
  4 
  5 ChoroplethJS.GmapOverlay.prototype.onAdd = function() {
  6 	var map;
  7 	var w = gmap.getDiv().scrollWidth;
  8 	var h = gmap.getDiv().scrollHeight;
  9 	this.div = document.createElement('div');
 10 	this.div.style.position = 'absolute';
 11 	this.div.style.overflow = 'visible';
 12 	this.div.style.top = 0;
 13 	this.div.style.left = 0;
 14 	this.div.style.width = w;
 15 	this.div.style.height = h;
 16 	this.div.id = 'gmapCanvas';
 17 	this.getPanes().overlayImage.appendChild(this.div);
 18 	
 19 	map = new ChoroplethJS('gmapCanvas', { projection: 'SphericalMercator' });
 20 	
 21 	this.scaleTo	= function(value, callback) { map.scaleTo(value, callback); };
 22 	this.reset		= function(value, callback) { map.reset(); };
 23 	this.setValue = function(key, value) { map.setValue(key, value); };
 24 	this.getShape = function(key) { return map.getShape(key); };
 25 	this.getShapeKeys = function() { return map.getShapeKeys(); };
 26 	
 27 };
 28 
 29 ChoroplethJS.GmapOverlay.prototype.draw = function() {
 30 	var center = this.getProjection().fromLatLngToDivPixel(new google.maps.LatLng(38, -97));
 31 	var topLeft = this.getProjection().fromLatLngToDivPixel(new google.maps.LatLng(74.03, -180));
 32 	this.scaleTo(this.getScaleByZoomLevel(gmap.getZoom()));
 33 	var x = topLeft.x;
 34 	var y = topLeft.y;
 35 	this.div.style.top = y + 'px';
 36 	this.div.style.left = x + 'px';
 37 };
 38 
 39 ChoroplethJS.GmapOverlay.prototype.onRemove = function() {
 40 	this.div.parentNode.removeChild(this.div);
 41 };
 42 
 43 ChoroplethJS.GmapOverlay.prototype.getScaleByZoomLevel = function(zoomLevel) {
 44 	var result = 1;
 45 	switch (zoomLevel) {
 46 		case 0: result = 0.0315; break;
 47 		case 1: result = 0.0625; break;
 48 		case 2: result = 0.125; break;
 49 		case 3: result = 0.25; break;
 50 		case 4: result = 0.5; break; 
 51 		case 5: break;
 52 		case 6: result = 2; break;
 53 		case 7: result = 4; break;
 54 		case 8: result = 8; break;
 55 		case 9: result = 16; break;
 56 		case 10: result = 32; break;
 57 		case 11: result = 64; break;
 58 		case 12: result = 128; break;
 59 		case 13: result = 256; break;
 60 		case 14: result = 512; break;
 61 		case 15: result = 1024; break;
 62 		default: break;
 63 	}
 64 	return result;
 65 };
